1.题目大意：

在n进制下，找出不超过m的质数x,满足x的所有前缀都是质数，求x的个数。

弱化版

2.思路：

首先要知道，一个符合要求的x,我们设它的最大前缀为a,则x=n*a+c (0≤c<n)。
知道这个之后，我们就可以使用宽搜。对于队列中的每一个数，先判断是否为质数，如果是，则先输出，再枚举c,将所有有可能的x放进队列；如果不是，则直接跳过。
由于题目说了答案小于等于2021，所以达到2021个时直接return即可。

3.代码：

#include<iostream>
#include<cstdio>
#include<queue>
using namespace std;
long long pri[2000000];
bool vis[5000001];
queue<long long>l;
//筛质数
void find(long long x)
{
    vis[1]=1;
    for(int i=2;i<=x;++i)
    {
        if(!vis[i])  pri[++pri[0]]=i;
        for(int j=1;j<=pri[0]&&i*pri[j]<=x;++j)
        {
            vis[i*pri[j]]=1;
            if(i%pri[j]==0)  break;
        }
    }
}
int n,qp;
long long m;
void bfs()
{
    while(l.size())
    {
        long long now=l.front();
        l.pop();
        bool p=1;
        //判断是否为质数
        for(register int i=2;pri[i]*pri[i]<=now;++i)  
        {
            if(now%pri[i]==0)
            {
                p=0;
                break;
            }
        }
        if(p)  printf("%lld ",now),qp++;
        else continue;
        //如果到了2021直接跳出。
        if(qp==2021)  return;

        for(register long long i=now*n+1;i<now*n+n;i++) 
        {
            if(i>m)  break;
            if(i%2==0)  continue;
            l.push(i);
        }
    }
}
int main()
{
    scanf("%d%lld",&n,&m);
    find(100000);
    //将所有在n进制下一位的质数扔进队列（不包括n）
    for(register int i=1;pri[i]<=m&&pri[i]<n;++i) l.push(pri[i]);
    bfs();
    return 0;
}
